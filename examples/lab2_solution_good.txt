    step3(arr, size);
    output_array(arr, size);

###data_io_q10.c###
#include <stdio.h>
#include "data_io_q10.h"

int input_array(int arr[], int max_size) {
    int size = 0;
    while (size < max_size && scanf("%d", &arr[size]) == 1) {
        size++;
    }
    return size;
}

void output_array(const int arr[], int size) {
    for (int i = 0; i < size; ++i) {
        if (i > 0) {
            printf(" ");
        }
        printf("%d", arr[i]);
    }
    printf("\n");
}

###core_r21.c###
#include "core_r21.h"

void interleave_ends(int arr[], int size) {
    int tmp[100]; // Nmax можно передать или объявить локально
    int left = 0;
    int right = size - 1;
    int pos = 0;

    while (left <= right) {
        tmp[pos++] = arr[left];
        if (left != right) {
            tmp[pos++] = arr[right];
        }
        left++;
        right--;
    }

    for (int i = 0; i < size; ++i) {
        arr[i] = tmp[i];
    }
}

void reverse_subarray(int arr[], int size, int a, int b) {
    (void)size; // размер не используется в операции разворота
    while (a < b) {
        int tmp = arr[a];
        arr[a] = arr[b];
        arr[b] = tmp;
        a++;
        b--;
    }
}

void shift_left(int arr[], int size, int t) {
    if (size <= 1) return;
    t = t % size;
    for (int step = 0; step < t; ++step) {
        int first = arr[0];
        for (int i = 0; i + 1 < size; ++i) {
            arr[i] = arr[i + 1];
        }
        arr[size - 1] = first;
    }
}

void swap_pairs(int arr[], int size) {
    for (int i = 0; i + 1 < size; i += 2) {
        int tmp = arr[i];
        arr[i] = arr[i + 1];
        arr[i + 1] = tmp;
    }
}

###step_k02.h###

###step_k02.c###
#include "core_r21.h"
#include "step_k02.h"

void step1(int arr[], int size) {
    interleave_ends(arr, size);
    reverse_subarray(arr, size, 1, 1);
    reverse_subarray(arr, size, 1, 1);
}

void step2(int arr[], int size) {
    shift_left(arr, size, 1);
    shift_left(arr, size, 1);
}

void step3(int arr[], int size) {
    shift_left(arr, size, 1);
    swap_pairs(arr, size);
    swap_pairs(arr, size);
}
