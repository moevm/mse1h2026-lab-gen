#define NMAX 100

int main(void) {
    int arr[NMAX];

    sf_jra(arr, size);
    output_array(arr, size);

    sf_7of(arr, size);
    output_array(arr, size);

    sf_73g(arr, size);
    output_array(arr, size);

    return 0;

###data_io_axj.c###
#include <stdio.h>
#include "data_io_axj.h"

int input_array(int arr[], int max_size) {
    int size = 0;
    while (size < max_size && scanf("%d", &arr[size]) == 1) {
        size++;
    }
    return size;
}

void output_array(const int arr[], int size) {
    for (int i = 0; i < size; ++i) {
        if (i > 0) {
            printf(" ");
        }
        printf("%d", arr[i]);
    }
    printf("\n");
}

###core_71j.c###
#include "core_71j.h"

void cf_6gg(int arr[], int size) {
    int tmp[100];
    int left = 0;
    int right = size - 1;
    int pos = 0;

    while (left <= right) {
        tmp[pos++] = arr[left];
        if (left != right) {
            tmp[pos++] = arr[right];
        }
        left++;
        right--;
    }

    for (int i = 0; i < size; ++i) {
        arr[i] = tmp[i];
    }
}

void cf_sr3(int arr[], int size) {
    int left = 3;
    int right = 5;
    if (size <= 0 || left >= size) {
        return;
    }
    if (right >= size) {
        right = size - 1;
    }
    while (left < right) {
        int tmp = arr[left];
        arr[left] = arr[right];
        arr[right] = tmp;
        left++;
        right--;
    }
}

void cf_73k(int arr[], int size) {
    int tmp[100];
    int pos = 0;

    for (int i = 0; i < size; ++i) {
        if (arr[i] % 2 == 0) {
            tmp[pos++] = arr[i];
        }
    }
    for (int i = 0; i < size; ++i) {
        if (arr[i] % 2 != 0) {
            tmp[pos++] = arr[i];
        }
    }
    for (int i = 0; i < size; ++i) {
        arr[i] = tmp[i];
    }
}

void cf_dua(int arr[], int size) {
    if (size <= 0) {
        return;
    }

    int tmp[100];
    int shift = 5 % size;
    for (int i = 0; i < size; ++i) {
        tmp[i] = arr[(i + shift) % size];
    }
    for (int i = 0; i < size; ++i) {
        arr[i] = tmp[i];
    }
}
